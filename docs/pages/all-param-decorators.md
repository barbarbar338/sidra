# All Param Decorators

-   @Body() -> req.body
-   @Headers(header?) -> req.headers || req.headers[header]
-   @Ip() -> req.ip
-   @Next() -> next function
-   @Param(param?) -> req.params || req.params[param];
-   @Query(query?) -> req.query || req.query[query];
-   @Res() -> res
-   @Req() -> req
